1)AS
    A) SELECT <column name> AS <alias> FROM <table name>;
    B) Example 
        i) SELECT username AS Username, first_name AS "First Name" FROM users;
        ii) SELECT title AS Title, year AS "Year Released" FROM movies;
        iii) SELECT name AS Name, description AS Description, price AS "Current Price" FROM products;
